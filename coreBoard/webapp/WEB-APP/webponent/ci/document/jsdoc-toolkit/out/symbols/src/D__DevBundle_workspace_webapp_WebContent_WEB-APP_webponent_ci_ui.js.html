 17 	function isElement(obj) {
 18 		try {
 19 			// Using W3 DOM2 (works for FF, Opera and Chrom)
 20 			return obj instanceof HTMLElement;
 21 		} catch (e) {
 22 			// Browsers not supporting W3 DOM2 don't have HTMLElement and
 23 			// an exception is thrown and we end up here. Testing some
 24 			// properties that all elements have. (works on IE7)
 25 			return (typeof obj === "object") && (obj.nodeType === 1) && (typeof obj.style === "object") && (typeof obj.ownerDocument === "object");
 26 		}
 27 	}

183 	 * @public
184 	 * @description object배열을 select의 option으로 세팅한다 (label은 화면에 보여질 값이며 나머지는
185 	 *              attribute로 들어간다.)
186 	 * @param option의
187 	 *            속성에 들어갈 데이터 Object를 담고있는 배열
188 	 * @param select
189 	 *            dom객체 or jquery객체, 문자
190 	 * @example ci.ui.setSelectOptions([{'label' : '레이블', 'value' : 'true',
191 	 *          'attr_1' : 'a', 'attr_2' : 'b' }], $('#select'))

195 	self.setSelectOptions = function(opsDataArr, selectObj) {
196 
197 		var opsHTML = "";
198 		for ( var i = 0; i < opsDataArr.length; i++) {
199 			var obj = opsDataArr[i];
200 			var label = "";
201 			opsHTML += "<option ";
202 			for ( var prop in obj) {
203 				if (prop == 'label') {
204 					label = obj[prop];
205 				} else {
206 					opsHTML += (prop + "='");
207 					opsHTML += (obj[prop] + "' ");
208 				}
209 			}
210 
211 			opsHTML += ">";
212 			opsHTML += label;
213 			opsHTML += "</option>";
214 		}
215 
216 		if (isJQueryObj(selectObj)) {
217 			selectObj.html(opsHTML);
218 		} else if (isString(selectObj)) {
219 			selectObj = document.getElementById(selectObj);
220 			selectObj.innerHTML = opsHTML;
221 		} else if (isElement(selectObj)) {
222 			selectObj.innerHTML = opsHTML;
223 		}
