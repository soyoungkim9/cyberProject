 13 (function(ci){
 14 
 15 	var self = {
 16 		//platform
 17 		os: {
 18 			name: '',
 19 			platform: '' //TODO
 20 		},

 22 		//renderer
 23 		renderer: {
 24 			name: '',
 25 			version: '',
 26 			isGecko: false,
 27 			isIe: false,
 28 			isOpera: false,
 29 			isWebkit: false
 30 		},
 31 
 32 		//browser
 33 		product: {
 34 			name: '', //browser name
 35 			version: '' //browser version
 36 		},
 37 
 38 		version: undefined,
 39 
 40 		isAndroid : false,
 41 		isCamino : false,
 42 		isChrome : false,
 43 		isFirefox : false,
 44 		isIe : false,
 45 		isIpad : false,
 46 		isIphone : false,
 47 		isOpera : false,
 48 		isSafari : false,
 49 		isMobile: false,
 50 
 51 		isLinux: false,
 52 		isWindows: false,
 53 		isMac: false,
 54 		isX11: false,
 55 
 56 		documnetMode: undefined, //IE specific
 57 
 58 		/**
 59 		 * ie일경우 ie7, ie8식으로 버전과함께 반환
 60 		 * 이외브라우져일경우 product명만 반환
 61 		 */
 62 		productVersion: '',
 63

 65 		 * 안전하게 CSS transform3D를 사용할수 있는지
 66 		 */
 67 		canUse3d : false

 70 	var userAgentString = window.navigator.userAgent;
 71 
 72 	function getURLParameter(name) {
 73 	    return decodeURI(
 74 	        (RegExp(name + '=' + '(.+?)(&|$)').exec(location.search)||[,null])[1]
 75 	    );
 76 	}
 77 
 78 	//debug mode
 79 	if (getURLParameter('device') == 'mobile') {
 80 		userAgentString = 'Mozilla/5.0 (iPhone; CPU iPhone OS 5_0 like Mac OS X) AppleWebKit/534.46 (KHTML, like Gecko) Version/5.1 Mobile/9A334 Safari/7534.48.3';
 81 	}
 82 
 83 	var navigator = window.navigator;
 84 	var platform = navigator && navigator.platform || '';
 85 
 86 	//utility functions
 87 	var contains = function(s, ss){
 88 		return s.indexOf(ss) != -1;
 89 	};
 90 
 91 	//initialize browser render and is mobile
 92 	var init = function(){
 93 		var ua = userAgentString;
 94 
 95 		self.renderer.isOpera = ua.indexOf('Opera') == 0;
 96 		self.renderer.isIe = !self.renderer.isOpera && contains(ua,'MSIE');
 97 		self.renderer.isWebkit = !self.renderer.isOpera && contains(ua,'WebKit');
 98 		self.isMobile = self.renderer.isWebkit && contains(ua,'Mobile');
 99 		self.renderer.isGecko = !self.renderer.isOpera && !self.renderer.isWebkit && navigator.product == 'Gecko';
100 
101 		if(self.renderer.isIe){
102 			var doc = document;
103 			self.documnetMode = doc ? doc['documentMode'] : undefined;
104 		}
105 
106 	};
107 
108 	init();
109 
110 	//determine browser version
111 	var determineRendererVersion = function(){
112 		var version = '', re;
113 
114 		if (self.renderer.isOpera) {
115 			var operaVersion = opera.version;
116 			version = typeof operaVersion == 'function' ? operaVersion() : operaVersion;
117 		} else {
118 			if (self.renderer.isGecko) {
119 				re = /rv\:([^\);]+)(\)|;)/;
120 			} else if (self.renderer.isIe) {
121 				re = /MSIE\s+([^\);]+)(\)|;)/;
122 			} else if (self.renderer.isWebkit) {
123 			  	// WebKit/125.4
124 				re = /WebKit\/(\S+)/;
125 			}
126 			if (re) {
127 				var arr = re.exec(userAgentString);
128 				version = arr ? arr[1] : '';
129 			}
130 		}
131 		// IE9 can be in document mode 9 but be reporting an inconsistent user agent
132 		// version.  If it is identifying as a version lower than 9 we take the
133 		// documentMode as the version instead.  IE8 has similar behavior.
134 		// It is recommended to set the X-UA-Compatible header to ensure that IE9
135 		// uses documentMode 9.
136 		if (self.renderer.isIe) {
137 			var docMode = self.documnetMode;
138 			if (docMode > parseFloat(version)) {
139 				return String(docMode);
140 			}
141 		}
142 		return version;
143 	};
144 
145 	self.renderer.version = determineRendererVersion();
146 
147 	//initialize os platform
148 	var initPlatform = function(){
149 		if(contains(platform, 'Win')){
150 			self.isWindows = true;
151 			self.os.name = 'windows';
152 		} else if(contains(platform, 'Mac')){
153 			self.isMac = true;
154 			self.os.name = 'mac';
155 		} else if(contains(platform, 'Linux')){
156 			self.isLinux = true;
157 			self.os.name = 'linux';
158 		} else if(contains(navigator['appVersion'] || '', 'X11')){
159 			self.isX11 = true;
160 			self.os.name = 'x11';
161 		}
162 	};
163 
164 	initPlatform();
165 
166 
167 	//initialize product
168 	var initProduct = function(){
169 		var ua = userAgentString;
170 
171 		if(self.renderer.isOpera){
172 			self.isOpera = true;
173 			self.product.name = 'opera';
174 		} else if (self.renderer.isIe) {
175 			self.isIe = true;
176 			self.product.name = 'ie';
177 		} else if (ua.indexOf('Firefox') != -1) {
178 			self.isFirefox = true;
179 			self.product.name = 'firefox';
180 		} else if (ua.indexOf('Camino') != -1) {
181 			self.isCamino = true;
182 			self.product.name = 'camino';
183 		} else if (ua.indexOf('iPhone') != -1 || ua.indexOf('iPod') != -1) {
184 			self.isIphone = true;
185 			self.product.name = 'iPhone';
186 		} else if (ua.indexOf('iPad') != -1) {
187 			self.isIpad = true;
188 			self.product.name = 'iPad';
189 		} else if (ua.indexOf('Android') != -1) {
190 			self.isAndroid = true;
191 			self.product.name = 'android';
192 		} else if (ua.indexOf('Chrome') != -1) {
193 			self.isChrome = true;
194 			self.product.name = 'chrome';
195 		} else if (ua.indexOf('Safari') != -1) {
196 			self.isSafari = true;
197 			self.product.name = 'safari';
198 		}
199 	};
200 
201 	initProduct();
202 
203 
204 	//initialize product version
205 	var initProductVersion = function(){
206 		if (self.isFirefox) {
207 			// Firefox/2.0.0.1 or Firefox/3.5.3
208 			return getFirstRegExpGroup(/Firefox\/([0-9.]+)/);
209 		}
210 		if (self.isIe || self.isOpera) {
211 			return self.renderer.version;
212 		}
213 		
214 		if (self.isChrome) {
215 			// Chrome/4.0.223.1
216 			return getFirstRegExpGroup(/Chrome\/([0-9.]+)/);
217 		}
218 		if (self.isSafari) {
219 			// Version/5.0.3
220 			//
221 			// NOTE: Before version 3, Safari did not report a product version number.
222 			// The product version number for these browsers will be the empty string.
223 			// They may be differentiated by WebKit version number in goog.userAgent.
224 			return getFirstRegExpGroup(/Version\/([0-9.]+)/);
225 		}
226 
227 		if (self.isIphone || self.isIpad) {
228 			// Mozilla/5.0 (iPod; U; CPU like Mac OS X; en) AppleWebKit/420.1
229 			// (KHTML, like Gecko) Version/3.0 Mobile/3A100a Safari/419.3
230 			// Version is the browser version, Mobile is the build number. We combine
231 			// the version string with the build number: 3.0.3A100a for the example.
232 			var arr = execRegExp(/Version\/(\S+).*Mobile\/(\S+)/);
233 			if (arr) {
234 				return arr[1] + '.' + arr[2];
235 			}
236 		} else if (self.isAndroid) {
237 			// Mozilla/5.0 (Linux; U; Android 0.5; en-us) AppleWebKit/522+
238 			// (KHTML, like Gecko) Safari/419.3
239 			//
240 			// Mozilla/5.0 (Linux; U; Android 1.0; en-us; dream) AppleWebKit/525.10+
241 			// (KHTML, like Gecko) Version/3.0.4 Mobile Safari/523.12.2
242 			//
243 			// Prefer Version number if present, else make do with the OS number
244 			var version = getFirstRegExpGroup(/Android\s+([0-9.]+)/);
245 			if (version) {
246 				return version;
247 			}
248 			return getFirstRegExpGroup(/Version\/([0-9.]+)/);
249 		} else if (self.isCamino) {
250 			return getFirstRegExpGroup(/Camino\/([0-9.]+)/);
251 		}
252 
253 		return '';
254 
255 		function getFirstRegExpGroup(re){
256 			var arr = execRegExp(re);
257   			return arr ? arr[1] : '';
258 		}
259 
260 		function execRegExp(re){
261 			return re.exec(userAgentString);
262 		}
263 	};
264 
265 	self.version = parseFloat(initProductVersion());
266 
267 	self.productVersion = 
268 			self.product.name === 'ie' ? self.product.name + self.version : self.product.name;
269 
270 	if ( self.isIphone || (self.isAndroid && self.version > 3)) {
271 		self.canUse3d = true;
272 	}
273 
274 	//add class to html element
275 	var addUserAgentClass = function(){
276 		var htmlElem = document.getElementsByTagName('html')[0];
277 		var classString = '';
278 
279 		classString += self.os.name === '' ? '' : self.os.name+' ';
280 		classString += self.product.name+' ';
281 		classString += self.product.name+self.version;
282 
283 		if(self.isMobile){
284 			classString += ' '+'mobile';
285 		}
286 
287 		htmlElem.className = classString;
288 	};
289 
290 	addUserAgentClass();
291 
292 	//regist self object to global scope
293 	if(!ci) window.ci = ci = {};
294 	ci.agent = self;

296 })(window.ci);
