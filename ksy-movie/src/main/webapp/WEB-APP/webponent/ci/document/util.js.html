 13 	function getByte(sChar) {
 14 		var c = 0;
 15 		var u = escape(sChar);
 16 		if (u.length < 4) { // 반각문자 : 기본적인 영문, 숫자, 특수기호
 17 			c++; // + 1byte
 18 		} else {
 19 			var s = parseInt(sChar.charCodeAt(0));
 20 			if (((s >= 65377) && (s <= 65500)) || ((s >= 65512) && (s <= 65518))) // 반각문자
 21 				// 유니코드
 22 				// 10진수
 23 				// 범위 :
 24 				// 한국어,
 25 				// 일본어,
 26 				// 특수문자
 27 				c++; // + 1byte
 28 			else
 29 				// 전각문자 : 위 조건을 제외한 모든 문자
 30 				c += 2; // + 2byte
 31 		}
 32 		return c;
 33 	}
